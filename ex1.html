//all includes
	
		
		class TextMemento {
		private:
			std::string text;
		
		public:
			TextMemento(const std::string& t) : text(t) {}
		
			std::string getText() const {
				return text;
			}
		};
		
		class TextEditor {
		private:
			std::string text;
			std::vector history;
			int currentIndex;
		
		public:
			TextEditor() : text(""), currentIndex(-1) {}
		
			void setText(const std::string& t) {
				text = t;
				history.push_back(TextMemento(text));
				currentIndex++;
			}
		
			std::string getText() const {
				return text;
			}
		
			void undo() {
				if (currentIndex > 0) {
					currentIndex--;
					text = history[currentIndex].getText();
				}
			}
		
			void redo() {
				if (currentIndex < history.size() - 1) {
					currentIndex++;
					text = history[currentIndex].getText();
				}
			}
		};
		
		int main() {
			TextEditor editor;
		
			editor.setText("This is the initial text.");
			std::cout << "Current text: " << editor.getText() << std::endl;
		
			editor.setText("Updated text.");
			std::cout << "Current text: " << editor.getText() << std::endl;
		
			editor.undo();
			std::cout << "Undo: " << editor.getText() << std::endl;
		
			editor.redo();
			std::cout << "Redo: " << editor.getText() << std::endl;
		
			return 0;
		}
